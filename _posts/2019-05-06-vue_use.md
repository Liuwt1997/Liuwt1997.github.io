### node.js

    node -v
    npm -v

### vue-cli

    npm install -g vue-cli
    vue init webpack Vue-Project
    npm run dev
  
### 新建空项目
    node -v npm -v
    npm install -g cnpm
    cnpm install -g vue
    cnpm install -g vue-cli //安装vue-cli 
    cnpm install -g webpack //安装webpack
    vue init webpack new-project    //新建项目    
    cnpm install    //安装依赖    
    cnpm run dev    //运行项目   
    cnpm run build  //发布项目
